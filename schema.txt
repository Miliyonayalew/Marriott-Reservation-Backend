rails g model User name:string 
rails g model Room name:string image:string description:text price:decimal type:string
rails g model Booking user:references room:references start_date:date end_date:date

class User < ApplicationRecord
  has_many :bookings, dependent: :destroy
  has_many :rooms, through: :bookings, dependent: :destroy

  validates :name, presence: true, length: { maximum: 40 }
end

class Room < ApplicationRecord
  has_many :bookings, dependent: :destroy
  has_many :users, through: :bookings, dependent: :destroy

  validates :name, presence: true, length: { maximum: 30 }
  validates :image, presence: true, length: { maximum: 30 }
  validates :name, presence: true, length: { maximum: 30 }
  validates :description, presence: true, length: { maximum: 400 }
  validates :price, presence: true, numericality: { greater_than: 0 }
  validates :name, presence: true, length: { maximum: 30 }
end

class Booking < ApplicationRecord
  belongs_to :user
  belongs_to :room

  validates :start_date, :end_date, presence: true
  validate :end_date_after_start_date
end
